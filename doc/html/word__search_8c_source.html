451 {

482  *
483  * @param word word to be inserted
484  * @param grid words grid where word will be inserted to
485  * @param grid_mask grid mask where inserted word will be marked
486  * @param generated_word pointer to place in array of
487  * generated words where word will be added

493  */
494 bool inserted_word(

498  Generated_word *generated_word,
499  Word_direction direction,

509  .direction = direction,
510  .order = order,
511  };
512

514  if (!(generated_word->word = (char *) malloc(word_length * sizeof(char))))
515  {
516  return false;

519 
520  // generates valid (not overlapping) word start coordinates
521  int row = 0, col = 0;
522  bool valid_coords;
523  do
524  {
525  valid_coords = true;
526 
527  switch (direction)
528  {
529  case WORD_DIRECTION_HORIZONTALLY:
530  row = (int) (random() % difficulty.grid_size);
531  col =
532  (int) (random() % (difficulty.grid_size - word_length + 1));
533  break;
534 
535  case WORD_DIRECTION_VERTICALLY:
536  row =
537  (int) (random() % (difficulty.grid_size - word_length + 1));
538  col = (int) (random() % difficulty.grid_size);
539  break;
540 
541  case WORD_DIRECTION_DIAGONALLY:
542  switch (order)
543  {
544  case WORD_ORDER_ASCENDING:
545  row = (int) (
546  random() % (difficulty.grid_size - word_length + 1)
547  );
548  col = (int) (
549  random() % (difficulty.grid_size - word_length + 1)
550  );
551  break;
552 
553  case WORD_ORDER_DESCENDING:
554  row = (int) (
555  word_length - 1 + random()
556  % (difficulty.grid_size - word_length + 1)
557  );
558  col = (int) (
559  word_length - 1 + random()
560  % (difficulty.grid_size - word_length + 1)
561  );
562  break;
563  }

582  generated_word->col = col;
583 
584  // insert generated word to grid and mark this word as

587  int i = 0, r = row, c = col;

589  i++, grid_move(direction, order, &r, &c)

780  if (new_line)
781  {

783  }
784

795  }
